# DB 拆分 0.1.2 → 0.2.0 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将单一 `ai_roll.db` 拆为 `roll-inner.db`（内在）+ `roll_outer.db`（外在），通过 SQLite ATTACH 连接。

**Architecture:** roll-inner.db 存 metadata/dna/loop/book/skill/history + 模板 pages/memory，构建后只读。roll_outer.db 存运行时 pages/memory/loop_runs。打开外层时 ATTACH 内层 AS inner，SQL 中用 `inner.` 前缀访问内表。system.db 新加 `outer_db_path` 列追踪 outer 路径。

**Tech Stack:** Go, SQLite (ATTACH DATABASE), Cobra CLI

- [ ] **Step 1: db.go — 新增 OpenOuter**

```go
// OpenOuter opens the outer database and attaches the inner database.
// All inner tables (metadata, dna, loop, book, skill, history, template pages/memory)
// are accessed with inner. prefix in SQL queries.
func OpenOuter(outerPath, innerPath string) (*sql.DB, error) {
    conn, err := Open(outerPath)
    if err != nil {
        return nil, err
    }
    _, err = conn.Exec("ATTACH DATABASE ? AS inner", innerPath)
    if err != nil {
        conn.Close()
        return nil, fmt.Errorf("attach inner: %w", err)
    }
    return conn, nil
}
```

```go
func resolvePageNewCwd(name, version string, args []string) (string, string, error) {
    var cwd string
    // ... existing cwd resolution logic ...
    
    // Determine outer db path
    var outerPath string
    var err error
    workspaceRoot, _ := store.IrollPath(name, version)
    workspacePath := filepath.Join(workspaceRoot, "workspace")
    workspaceAbs, _ := filepath.Abs(workspacePath)
    
    if pageNewCwd != "" {
        cwd, _ = filepath.Abs(pageNewCwd)
    } else if len(args) > 1 {
        cwd, _ = filepath.Abs(args[1])
    }
    // else default workspace
    
    if cwd == "" || cwd == workspaceAbs {
        // Default workspace mode
        outerPath, err = store.WorkspaceOuterDbPath(name, version)
        if err != nil {
            return "", "", err
        }
        if cwd == "" {
            cwd = workspaceAbs
        }
        os.MkdirAll(workspaceAbs, 0755)
    } else {
        // Custom cwd mode
        outerPath, err = store.CwdOuterDbPath(cwd, name)
        if err != nil {
            return "", "", err
        }
        os.MkdirAll(filepath.Dir(outerPath), 0755)
    }
    
    // Copy outer template if not exists
    if _, err := os.Stat(outerPath); os.IsNotExist(err) {
        irollDir := checkedInnerPath(name, version)
        templateOuter := filepath.Join(filepath.Dir(irollDir), "roll-outer.db")
        if err := copyFile(templateOuter, outerPath); err != nil {
            return "", "", fmt.Errorf("copy outer db template: %w", err)
        }
    }
    
    return cwd, outerPath, nil
}
```

```go
func openTestOuter(t *testing.T) (*sql.DB, func()) {

    dir := t.TempDir()
    innerPath := filepath.Join(dir, "roll-inner.db")
    outerPath := filepath.Join(dir, "roll-outer.db")

    schema, _ := os.ReadFile(filepath.Join("..", "..", "examples", "base-agent", "init_inner.sql"))

}
```
