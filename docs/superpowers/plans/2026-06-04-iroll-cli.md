**Goal:** 构建 Go CLI 工具 `iroll`，使用 Cobra 框架管理 .iroll 包（加载、session 管理、context 读写、记忆新增）

**Tech Stack:** Go 1.24, Cobra, go-sqlite3 (CGO), google/uuid

	if err != nil {
		outputError(fmt.Sprintf("JSON marshal error: %v", err))

}

	b, _ := json.Marshal(map[string]string{"error": msg})
	fmt.Fprintln(os.Stderr, string(b))
	os.Exit(1)
}

```go
package store

import (
	"archive/zip"
	"database/sql"
	"fmt"
	"os"
	"path/filepath"

	_ "github.com/mattn/go-sqlite3"
)

func HomeDir() string {
	home, _ := os.UserHomeDir()
	return filepath.Join(home, ".iroll")
}

func IrollPath(name string) string {
	return filepath.Join(HomeDir(), name)
}

func DbPath(name string) string {
	return filepath.Join(IrollPath(name), "ai_roll.db")
}

// ReadName 从 ZIP 中的 ai_roll.db 读取 metadata 表的 name 值
func ReadName(zipPath string) (string, error) {
	r, err := zip.OpenReader(zipPath)
	if err != nil {
		return "", fmt.Errorf("open zip: %w", err)
	}
	defer r.Close()

	tmpDir, err := os.MkdirTemp("", "iroll-load-*")
	if err != nil {
		return "", err
	}
	defer os.RemoveAll(tmpDir)

	for _, f := range r.File {
		if f.Name == "ai_roll.db" {
			rc, err := f.Open()
			if err != nil {
				return "", err
			}
			outPath := filepath.Join(tmpDir, "ai_roll.db")
			outFile, err := os.Create(outPath)
			if err != nil {
				rc.Close()
				return "", err
			}
			if _, err = outFile.ReadFrom(rc); err != nil {
				outFile.Close()
				rc.Close()
				return "", err
			}
			outFile.Close()
			rc.Close()

			db, err := sql.Open("sqlite3", outPath)
			if err != nil {
				return "", err
			}
			defer db.Close()

			var name string
			err = db.QueryRow("SELECT value FROM metadata WHERE key = 'name'").Scan(&name)
			if err != nil {
				return "", fmt.Errorf("read name from metadata: %w", err)
			}
			return name, nil
		}
	}

	return "", fmt.Errorf("ai_roll.db not found in zip")
}

// Extract 将 .iroll ZIP 解压到 ~/.iroll/<name>/
func Extract(zipPath string, name string) error {
	dest := IrollPath(name)
	if _, err := os.Stat(dest); err == nil {
		return fmt.Errorf("iroll '%s' already exists", name)
	}

	r, err := zip.OpenReader(zipPath)
	if err != nil {
		return fmt.Errorf("open zip: %w", err)
	}
	defer r.Close()

	os.MkdirAll(dest, 0755)

	for _, f := range r.File {
		outPath := filepath.Join(dest, f.Name)

		if f.FileInfo().IsDir() {
			os.MkdirAll(outPath, 0755)
			continue
		}

		if err := os.MkdirAll(filepath.Dir(outPath), 0755); err != nil {
			return err
		}

		outFile, err := os.Create(outPath)
		if err != nil {
			return err
		}

		rc, err := f.Open()
		if err != nil {
			outFile.Close()
			return err
		}

		_, err = outFile.ReadFrom(rc)
		rc.Close()
		outFile.Close()
		if err != nil {
			return err
		}
	}

	return nil
}

// List 返回所有已加载的 iroll 名称
func List() ([]string, error) {
	home := HomeDir()
	entries, err := os.ReadDir(home)
	if err != nil {
		if os.IsNotExist(err) {
			return nil, nil
		}
		return nil, err
	}

	var names []string
	for _, e := range entries {
		if e.IsDir() {
			dbFile := filepath.Join(home, e.Name(), "ai_roll.db")
			if _, err := os.Stat(dbFile); err == nil {
				names = append(names, e.Name())
			}
		}
	}
	return names, nil
}
```

```go
package db

import (
	"database/sql"
	"fmt"
	"time"

	_ "github.com/mattn/go-sqlite3"

	"github.com/google/uuid"
)

type Context struct {
	ID        int    `json:"id"`
	SessionID string `json:"session_id"`
	Cwd       string `json:"cwd"`
	Content   string `json:"content"`
	CreatedAt string `json:"created_at"`
	UpdatedAt string `json:"updated_at"`
}

type Memory struct {
	ID         int     `json:"id"`
	Content    string  `json:"content"`
	CreatedAt  string  `json:"created_at"`
	Importance float64 `json:"importance"`
}

func nowISO() string {
	return time.Now().UTC().Format(time.RFC3339Nano)
}

func Open(dbPath string) (*sql.DB, error) {
	return sql.Open("sqlite3", dbPath)
}

func InsertContext(db *sql.DB, cwd string) (*Context, error) {
	sessionID := uuid.New().String()
	now := nowISO()

	res, err := db.Exec(
		"INSERT INTO context (session_id, cwd, content, created_at, updated_at) VALUES (?, ?, '', ?, ?)",
		sessionID, cwd, now, now,
	)
	if err != nil {
		return nil, fmt.Errorf("insert context: %w", err)
	}

	id, _ := res.LastInsertId()
	return &Context{
		ID:        int(id),
		SessionID: sessionID,
		Cwd:       cwd,
		Content:   "",
		CreatedAt: now,
		UpdatedAt: now,
	}, nil
}

func ListContextsByCwd(db *sql.DB, cwd string) ([]Context, error) {
	rows, err := db.Query(
		"SELECT id, session_id, cwd, content, created_at, updated_at FROM context WHERE cwd = ?",
		cwd,
	)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var result []Context
	for rows.Next() {
		var c Context
		if err := rows.Scan(&c.ID, &c.SessionID, &c.Cwd, &c.Content, &c.CreatedAt, &c.UpdatedAt); err != nil {
			return nil, err
		}
		result = append(result, c)
	}
	return result, nil
}

func GetContextBySessionID(db *sql.DB, sessionID string) (*Context, error) {
	var c Context
	err := db.QueryRow(
		"SELECT id, session_id, cwd, content, created_at, updated_at FROM context WHERE session_id = ?",
		sessionID,
	).Scan(&c.ID, &c.SessionID, &c.Cwd, &c.Content, &c.CreatedAt, &c.UpdatedAt)
	if err != nil {
		return nil, fmt.Errorf("context not found: %w", err)
	}
	return &c, nil
}

func UpdateContextContent(db *sql.DB, sessionID string, content string) (*Context, error) {
	now := nowISO()
	_, err := db.Exec(
		"UPDATE context SET content = ?, updated_at = ? WHERE session_id = ?",
		content, now, sessionID,
	)
	if err != nil {
		return nil, fmt.Errorf("update context: %w", err)
	}
	return GetContextBySessionID(db, sessionID)
}

func InsertMemory(db *sql.DB, content string, importance float64) (*Memory, error) {
	now := nowISO()

	res, err := db.Exec(
		"INSERT INTO memory (content, created_at, importance) VALUES (?, ?, ?)",
		content, now, importance,
	)
	if err != nil {
		return nil, fmt.Errorf("insert memory: %w", err)
	}

	id, _ := res.LastInsertId()
	return &Memory{
		ID:         int(id),
		Content:    content,
		CreatedAt:  now,
		Importance: importance,
	}, nil
}
```

package cmd

import (
	"iroll/db"

	"github.com/spf13/cobra"

	Use:   "session",
	Short: "Manage sessions",

var sessionInitCwd string

var sessionListCmd = &cobra.Command{
	Use:   "list <name>",
	Short: "List sessions by cwd",
	Args:  cobra.ExactArgs(1),
	Run: func(cmd *cobra.Command, args []string) {
		name := args[0]
		conn, err := db.Open(store.DbPath(name))
		if err != nil {
			outputError(err.Error())
		}
		defer conn.Close()

		contexts, err := db.ListContextsByCwd(conn, sessionListCwd)
		if err != nil {
			outputError(err.Error())
		}

		if contexts == nil {
			contexts = []db.Context{}
		}
		outputJSON(contexts)
	},

var sessionInitCmd = &cobra.Command{
	Use:   "init <name>",
	Short: "Create a new session",
	Args:  cobra.ExactArgs(1),

		name := args[0]
		conn, err := db.Open(store.DbPath(name))
		if err != nil {
			outputError(err.Error())
		}
		defer conn.Close()

		ctx, err := db.InsertContext(conn, sessionInitCwd)
		if err != nil {
			outputError(err.Error())
		}

		outputJSON(ctx)
	},

	sessionListCmd.Flags().StringVar(&sessionListCwd, "cwd", "", "Working directory to filter by")
