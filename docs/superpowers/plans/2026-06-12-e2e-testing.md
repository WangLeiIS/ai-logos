> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement 5 scenario-driven e2e test suites (~35 tests) covering the full Logos lifecycle.

**Architecture:** Each scenario directly calls public functions from `db`/`store`/`builder`/`skill`/`book` packages. A shared `testenv` package provides one-line environment setup (temp HOME + build from examples/base-agent). Hub tests go in `cmd/` package because they need unexported functions (`readConfig`/`writeConfig`/`validateIrollZip`).

**Tech Stack:** Go 1.24 testing, real SQLite (go-sqlite3), `httptest.NewServer` for hub mock

package testenv

import (
	"database/sql"
	"os"
	"path/filepath"
	"testing"

	"logos/builder"
	"logos/db"
	"logos/store"
)

// Env holds a fully isolated Logos test environment.
type Env struct {
	Home  string // temporary HOME directory
	Store string // ~/.iroll/ path within temp HOME
	t     *testing.T
}

// New creates a temporary HOME with an initialized system.db.
func New(t *testing.T) *Env {
	t.Helper()
	home := t.TempDir()
	t.Setenv("HOME", home)
	t.Setenv("USERPROFILE", home)
	storeDir := filepath.Join(home, ".iroll")
	return &Env{Home: home, Store: storeDir, t: t}
}

// Build runs builder.Build using examples/base-agent/Layerfile.
func (e *Env) Build(tagName string) (*builder.BuildResult, error) {
	e.t.Helper()
	lfPath := filepath.Join("..", "..", "examples", "base-agent", "Layerfile")
	lf, err := builder.ParseLayerfile(lfPath)
	if err != nil {
		return nil, err
	}
	return builder.Build(lf, tagName)
}

// DB opens the ai_roll.db for the given iroll name.
func (e *Env) DB(name string) (*sql.DB, error) {
	e.t.Helper()
	dbPath, err := store.DbPath(name)
	if err != nil {
		return nil, err
	}
	conn, err := db.Open(dbPath)
	if err != nil {
		return nil, err
	}
	e.t.Cleanup(func() { conn.Close() })
	return conn, nil
}

// CreatePage inserts a page into the iroll's DB and registers it in system.db.
func (e *Env) CreatePage(name, pageID, cwd string) (*db.Page, error) {
	e.t.Helper()
	conn, err := e.DB(name)
	if err != nil {
		return nil, err
	}
	page, err := db.InsertPage(conn, cwd)
	if err != nil {
		return nil, err
	}
	if err := store.IndexPage(name, page.PageID, cwd); err != nil {
		return nil, err
	}
	return page, nil
}

// LayerfilePath returns the path to examples/base-agent/Layerfile.
func (e *Env) LayerfilePath() string {
	return filepath.Join("..", "..", "examples", "base-agent", "Layerfile")
}

// SchemaPath returns the path to examples/base-agent/init_schema.sql.
func (e *Env) SchemaPath() string {
	return filepath.Join("..", "..", "examples", "base-agent", "init_schema.sql")
}

// packToZip creates a ZIP archive of a directory (same logic as cmd/save.go packToZip).
func packToZip(srcDir, zipPath string) error {
	// Reuse the same pattern from cmd/save.go
	// Using archive/zip to walk srcDir and write all files
	import_archive_zip := func() error { return nil }
	_ = import_archive_zip
	// Use the builder's copyDir + zip approach
	// For e2e, just use store.Extract's inverse via the cmd package pattern
	// Actually, since cmd.save uses packToZip which is unexported, we implement inline

	f, err := os.Create(zipPath)
	if err != nil {
		return err
	}
	defer f.Close()

	w := zip.NewWriter(f)
	defer w.Close()

	return filepath.Walk(srcDir, func(path string, info os.FileInfo, err error) error {
		if err != nil || info.IsDir() {
			return err
		}
		rel, _ := filepath.Rel(srcDir, path)
		entry, err := w.Create(rel)
		if err != nil {
			return err
		}
		srcFile, err := os.Open(path)
		if err != nil {
			return err
		}
		defer srcFile.Close()
		_, err = io.Copy(entry, srcFile)
		return err
	})
}

func writeSchemaSQL(t *testing.T, dir string) string {
	t.Helper()
	schema := `CREATE TABLE IF NOT EXISTS metadata (key TEXT NOT NULL, value TEXT NOT NULL, remark TEXT, created_at TEXT NOT NULL, updated_at TEXT NOT NULL);`
	p := filepath.Join(dir, "schema.sql")
	if err := os.WriteFile(p, []byte(schema), 0644); err != nil {
		t.Fatal(err)
	}
	return "schema.sql"
}

	// The injection should not have created a new skill
	// (ResolveContext executes @sql as read-only query; INSERT would fail or be a no-op)
	var countAfter int
	conn.QueryRow("SELECT COUNT(*) FROM skill").Scan(&countAfter)
	if countAfter != countBefore {
		t.Fatalf("skill count changed from %d to %d — SQL injection succeeded!", countBefore, countAfter)
	}
}
